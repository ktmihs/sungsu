int main() {

  string hobbies[3] = {"누워있기", "카톡하기", "보리 괴롭히기"};

  cout << "Hobbies" << endl;

  }

  cout << "Specialties" << endl;;
  for (int i = 0; i < 2; ++i) {
    cout << "- " << specialties[i] << endl;
  }
